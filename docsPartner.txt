=== KONTEKS GISTREAM API ===

GiStream adalah platform streaming JKT48. Untuk memutar stream,
flow-nya adalah:

  1. Backend generate JWT token dari Token API (HMAC proof-of-knowledge)
  2. Backend / frontend ambil stream URL dari Stream Worker
  3. Frontend/player pakai token di header untuk HLS proxy

BASE URLs:
  Token API    : https://v5.jkt48connect.com
  Stream Worker: https://ctv.jkt48connect.com
  HLS Proxy    : https://stp33.jkt48connect.com

=== CREDENTIAL ===

Saya punya dua credential dari GiStream:
  PARTNER_KID    = "jkt48connect-v1"   ← non-rahasia, boleh di kode
  PARTNER_SECRET = <dari env var>       ← HANYA di server, tidak pernah dikirim

=== CARA KERJA AUTH (HMAC Proof-of-Knowledge) ===

Tidak ada header x-partner-secret. Secret dibuktikan lewat HMAC:

  signingPath  = "/api/token/generate?apikey=JKTCONNECT"
  bodyForHash  = "{}"   ← selalu string ini

  signingString = timestamp + ":" + nonce + ":" + "POST" + ":" + signingPath + ":" + SHA256(bodyForHash)
  signature     = HMAC-SHA256(signingString, PARTNER_SECRET)  → hex string

  timestamp = Date.now() dalam MILLISECONDS
  nonce     = random UUID tanpa strip, sekali pakai

Request headers ke Token API:
  x-kid        : PARTNER_KID
  x-timestamp  : timestamp (ms)
  x-nonce      : nonce unik
  x-signature  : HMAC hex
  x-showid     : show ID (atau x-slug jika pakai slug)
  Content-Type : application/json
  Body         : {}

TIDAK ADA x-partner-secret di header.

=== ATURAN WAJIB ===

1. PARTNER_SECRET hanya di server (env var), tidak pernah ke client
2. x-kid boleh ada di kode (non-rahasia)
3. Generate token HANYA di backend — tidak boleh di browser/RN
4. Token bersifat one-time-use saat hit /stream — jangan panggil dua kali
5. Frontend inject token via header x-api-token ke HLS proxy (stp33)
6. Lindungi endpoint generate token dengan autentikasi user kamu

=== AUTH MECHANISM — HMAC PROOF-OF-KNOWLEDGE ===

Setiap request ke Token API menggunakan HMAC signature sebagai
bukti bahwa server tahu PARTNER_SECRET — secret tidak pernah
dikirim di header.

Formula:
  body_hash    = SHA256("{}")  ← selalu hash dari string "{}"
  signing_str  = timestamp + ":" + nonce + ":" + "POST" +
                 ":" + "/api/token/generate?apikey=JKTCONNECT" +
                 ":" + body_hash
  signature    = HMAC-SHA256(signing_str, PARTNER_SECRET) → hex

Headers yang dikirim ke Token API:
  x-kid        : "jkt48connect-v1"
  x-timestamp  : Date.now() dalam ms (string)
  x-nonce      : UUID tanpa strip, sekali pakai per 60 detik
  x-signature  : signature hex
  x-showid     : show ID yang ingin diputar (atau x-slug)
  Content-Type : application/json
  Body         : {}

TIDAK ADA header x-partner-secret.

--- Node.js / TypeScript (untuk Prompt A & B) ---

const PARTNER_KID    = "jkt48connect-v1";
const PARTNER_SECRET = process.env.PARTNER_SECRET!;
const SIGNING_PATH   = "/api/token/generate?apikey=JKTCONNECT";

async function sha256Hex(str: string) {
  const buf = await crypto.subtle.digest("SHA-256", new TextEncoder().encode(str));
  return Array.from(new Uint8Array(buf)).map(b => b.toString(16).padStart(2,"0")).join("");
}

async function hmacHex(secret: string, msg: string) {
  const key = await crypto.subtle.importKey(
    "raw", new TextEncoder().encode(secret),
    { name: "HMAC", hash: "SHA-256" }, false, ["sign"]
  );
  const buf = await crypto.subtle.sign("HMAC", key, new TextEncoder().encode(msg));
  return Array.from(new Uint8Array(buf)).map(b => b.toString(16).padStart(2,"0")).join("");
}

async function buildHMACHeaders() {
  const ts   = Date.now().toString();
  const nc   = crypto.randomUUID().replace(/-/g,"");
  const bh   = await sha256Hex("{}");
  const sig  = await hmacHex(PARTNER_SECRET,
    `${ts}:${nc}:POST:${SIGNING_PATH}:${bh}`);
  return { "x-kid": PARTNER_KID, "x-timestamp": ts,
           "x-nonce": nc, "x-signature": sig };
}

async function generateStreamToken(showId: string, isSlug = false) {
  const h = await buildHMACHeaders();
  const res = await fetch(`https://v5.jkt48connect.com${SIGNING_PATH}`, {
    method: "POST",
    headers: { ...h,
      ...(isSlug ? {"x-slug": showId} : {"x-showid": showId}),
      "Content-Type": "application/json" },
    body: "{}",
  });
  const data = await res.json();
  if (!data.status) throw new Error(data.message);
  return data.data.token as string;
}

async function getStreamURL(token: string, showId: string, isSlug = false) {
  const p = isSlug ? `slug=${showId}` : `showId=${showId}`;
  const res = await fetch(`https://ctv.jkt48connect.com/stream?${p}`, {
    headers: { "x-api-token": token,
      ...(isSlug ? {"x-slug": showId} : {"x-showid": showId}) },
  });
  const data = await res.json();
  if (!data.success) throw new Error(data.message);
  return data.streams as any[];
}

// Hook ini dipanggil oleh komponen frontend.
// Ia fetch token dari BACKEND KAMU — bukan langsung ke GiStream.
function useStreamToken(backendUrl: string, showId: string) {
  const [token,   setToken]   = React.useState<string | null>(null);
  const [streams, setStreams] = React.useState<any[]>([]);
  const [loading, setLoading] = React.useState(true);
  const [error,   setError]   = React.useState<string | null>(null);

  const fetchToken = React.useCallback(async () => {
    setLoading(true); setError(null);
    try {
      const res = await fetch(`${backendUrl}/api/stream-token?showId=${showId}`);
      if (!res.ok) throw new Error(`HTTP ${res.status}`);
      const data = await res.json();
      setToken(data.token);          // simpan di state, BUKAN localStorage
      setStreams(data.streams || []);
    } catch (e: any) {
      setError(e.message);
    } finally {
      setLoading(false);
    }
  }, [backendUrl, showId]);

  React.useEffect(() => { fetchToken(); }, [fetchToken]);

  // Auto-refresh 5 menit sebelum expired
  React.useEffect(() => {
    if (!token) return;
    const timer = setInterval(() => {
      try {
        const exp = JSON.parse(atob(token.split(".")[1]
          .replace(/-/g,"+").replace(/_/g,"/"))).exp;
        if (exp - Date.now()/1000 <= 300) fetchToken();
      } catch {}
    }, 5 * 60 * 1000);
    return () => clearInterval(timer);
  }, [token, fetchToken]);

  return { token, streams, loading, error, refresh: fetchToken };
}
